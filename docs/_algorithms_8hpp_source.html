56 template <typename ForwardIterator, typename T>
57 ForwardIterator binarySearch(ForwardIterator first, ForwardIterator last, const T& value)
58 {
59  // Note: std::lower_bound() has O(log n) on random access iterators
60  ForwardIterator result = std::lower_bound(first, last, value);
61 
62  // std::lower_bound() returns iterator to first element >= value, which can be inequal to value
63  if (result == last || !equivalent(*result, value))
64  return last;
65  else
66  return result;
67 }
