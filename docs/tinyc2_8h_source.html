87 /*
88  DETAILS/ADVICE:
89  This header does not implement a broad-phase, and instead concerns itself with
90  the narrow-phase. This means this header just checks to see if two individual
91  shapes are touching, and can give information about how they are touching.
92 
93  Very common 2D broad-phases are tree and grid approaches. Quad trees are good
94  for static geometry that does not move much if at all. Dynamic AABB trees are
95  good for general purpose use, and can handle moving objects very well. Grids
96  are great and are similar to quad trees.
97 
98  If implementing a grid it can be wise to have each collideable grid cell hold
99  an integer. This integer refers to a 2D shape that can be passed into the
100  various functions in this header. The shape can be transformed from "model"
101  space to "world" space using c2x -- a transform struct. In this way a grid
102  can be implemented that holds any kind of convex shape (that this header
103  supports) while conserving memory with shape instancing.
104 
105  Please email at my address with any questions or comments at:
106  author's last name followed by 1748 at gmail
107 */

686 {
687  c2v a = s->a.p;

694  float vBC = c2Dot( b, c2Norm( c2Sub( b, c ) ) );

703  {
704  s->a.u = 1.0f;
705  s->div = 1.0f;
706  s->count = 1;
707  }

712  s->a.u = 1.0f;
713  s->div = 1.0f;
714  s->count = 1;
715  }
716 
717  else if ( uBC <= 0 && vCA <= 0 )
718  {
719  s->a = s->c;
720  s->a.u = 1.0f;
721  s->div = 1.0f;
722  s->count = 1;
723  }
724

728  s->b.u = vAB;

730  s->count = 2;
731  }

736  s->b = s->c;
737  s->a.u = uBC;
738  s->b.u = vBC;
739  s->div = uBC + vBC;
740  s->count = 2;
741  }
742

744  {

746  s->a = s->c;
747  s->a.u = uCA;

765 // Please see http://box2d.org/downloads/ under GDC 2010 for Erin's demo code
766 // and PDF slides for documentation on the GJK algorithm.
767 float c2GJK( const void* A, C2_TYPE typeA, const c2x* ax_ptr, const void* B, C2_TYPE typeB, const c2x* bx_ptr, c2v* outA, c2v* outB, int use_radius )
768 {
769  c2x ax;
770  c2x bx;
771  if ( typeA != C2_POLY || !ax_ptr ) ax = c2xIdentity( );
772  else ax = *ax_ptr;
773  if ( typeB != C2_POLY || !bx_ptr ) bx = c2xIdentity( );
774  else bx = *bx_ptr;
775 
776  c2Proxy pA;
777  c2Proxy pB;
778  c2MakeProxy( A, typeA, &pA );
779  c2MakeProxy( B, typeB, &pB );
780 
781  c2Simplex s;
782  s.a.iA = 0;
783  s.a.iB = 0;
784  s.a.sA = c2Mulxv( ax, pA.verts[ 0 ] );
785  s.a.sB = c2Mulxv( bx, pB.verts[ 0 ] );
786  s.a.p = c2Sub( s.a.sB, s.a.sA );
787  s.a.u = 1.0f;
788  s.count = 1;
789 
790  c2sv* verts = &s.a;
791  int saveA[ 3 ], saveB[ 3 ];
792  int save_count = 0;
793  float d0 = FLT_MAX;
794  float d1 = FLT_MAX;
795  int iter = 0;
796  int hit = 0;
797  while ( iter < C2_GJK_ITERS )
798  {
799  save_count = s.count;
800  for ( int i = 0; i < save_count; ++i )
801  {
802  saveA[ i ] = verts[ i ].iA;
803  saveB[ i ] = verts[ i ].iB;
804  }
805 
806  switch ( s.count )
807  {
808  case 1: break;
809  case 2: c22( &s ); break;
810  case 3: c23( &s ); break;
811  }
812 
813  if ( s.count == 3 )
814  {
815  hit = 1;
816  break;
817  }
818 
819  c2v p = c2L( &s );
820  d1 = c2Dot( p, p );
821 
822  if ( d1 > d0 ) break;
823  d0 = d1;
824 
825  c2v d = c2D( &s );
826  if ( c2Dot( d, d ) < FLT_EPSILON * FLT_EPSILON ) break;
827 
828  int iA = c2Support( pA.verts, pA.count, c2MulrvT( ax.r, c2Neg( d ) ) );
829  c2v sA = c2Mulxv( ax, pA.verts[ iA ] );
830  int iB = c2Support( pB.verts, pB.count, c2MulrvT( bx.r, d ) );
831  c2v sB = c2Mulxv( bx, pB.verts[ iB ] );
832 
833  ++iter;
834 
835  int dup = 0;
836  for ( int i = 0; i < save_count; ++i )
837  {
838  if ( iA == saveA[ i ] && iB == saveB[ i ] )
839  {
840  dup = 1;
841  break;
842  }
843  }
844  if ( dup ) break;
845 
846  c2sv* v = verts + s.count;
847  v->iA = iA;
848  v->sA = sA;
849  v->iB = iB;
850  v->sB = sB;
851  v->p = c2Sub( v->sB, v->sA );
852  ++s.count;
853  }
854 
855  c2v a, b;
856  c2Witness( &s, &a, &b );
857  float dist = c2Len( c2Sub( a, b ) );
858 
859  if ( hit )
860  {
861  a = b;
862  dist = 0;
863  }
864 
865  else if ( use_radius )
866  {
867  float rA = pA.radius;
868  float rB = pB.radius;
869 
870  if ( dist > rA + rB && dist > FLT_EPSILON )
871  {
872  dist -= rA + rB;
873  c2v n = c2Norm( c2Sub( b, a ) );
874  a = c2Add( a, c2Mulvs( n, rA ) );
875  b = c2Sub( b, c2Mulvs( n, rB ) );
876  }
877 
878  else
879  {
880  c2v p = c2Mulvs( c2Add( a, b ), 0.5f );
881  a = p;
882  b = p;
883  dist = 0;
884  }
885  }
886 
887  if ( outA ) *outA = a;
888  if ( outB ) *outB = b;
889  return dist;
890 }

1629 // for a full-algorithm overview. The short description is:
1630  // Test A against faces of B, test B against faces of A
1631  // Define the reference and incident shapes (denoted by r and i respectively)
1632  // Define the reference face as the axis of minimum penetration
1633  // Find the incident face, which is most anti-normal face
1634  // Clip incident face to reference face side planes

1637 {
1638  m->count = 0;
1639  c2x ax = ax_ptr ? *ax_ptr : c2xIdentity( );
1640  c2x bx = bx_ptr ? *bx_ptr : c2xIdentity( );
1641  int ea, eb;
1642  float sa, sb;
1643  if ( (sa = c2CheckFaces( A, ax, B, bx, &ea )) >= 0 ) return;
1644  if ( (sb = c2CheckFaces( B, bx, A, ax, &eb )) >= 0 ) return;
1645 
1646  const c2Poly* rp,* ip;
1647  c2x rx, ix;
1648  int re;
1649  float kRelTol = 0.95f, kAbsTol = 0.01f;
1650  int flip;
1651  if ( sa * kRelTol > sb + kAbsTol )
1652  {
1653  rp = A; rx = ax;
1654  ip = B; ix = bx;
1655  re = ea;
1656  flip = 0;
1657  }
1658  else
1659  {
1660  rp = B; rx = bx;
1661  ip = A; ix = ax;
1662  re = eb;
1663  flip = 1;
1664  }
