45  // Class to store a unique command listener.
46  // When a Listener instance is destroyed, all weak pointers from Connections to it are invalidated,
47  // thus preventing undefined behavior when disconnecting destroyed Listeners.
48  template <typename Parameter>
49  class Listener
50  {
51  private:
52  // Callback function type
53  typedef std::function<void(Parameter)> Function;
54 
55  public:
56  // Constructor, initializes function with fn and creates a new unique id
57  Listener(const Function& fn)
58  : mFunction(fn)
59  , mStrongRef() // will be initialized later by setEnvironment()
60  {
61  }
62 
63  // Call function
64  void call(Parameter arg) const
65  {
66  mFunction(arg);
67  }
68 
69  // Swap instances
70  void swap(Listener& other)
71  {
72  std::swap(mFunction, other.mFunction);
73  std::swap(mStrongRef, other.mStrongRef);
74  }
75 
76  // Sets the container and iterator in which this Listener is hold (to be able to create
77  // Connections from *this)
78  template <class ListenerContainer>
79  void setEnvironment(ListenerContainer& container, typename ListenerContainer::Iterator iterator)
80  {
81  mStrongRef = makeIteratorConnectionImpl(container, iterator);
82  }
83 
84  // Creates a Connection that can disconnect this Listener
85  Connection shareConnection() const
86  {
87  return Connection(mStrongRef);
88  }
89 
90  private:
91  Function mFunction;
92  std::shared_ptr<AbstractConnectionImpl> mStrongRef;
93  };

96  // Sequential container for listener
97  template <typename Parameter>
98  class ListenerSequence
99  {
100  public:
101  // The type of the function together with the id
102  typedef Listener<Parameter> ValueType;
103 
104  private:
105  // The container type used to store the callback functions
106  typedef std::vector<ValueType> Container;
107 
108  public:
109  // The iterator type (used to disconnect listeners)
110  typedef typename Container::iterator Iterator;
111 
112  public:
113  // Inserts a new listener to the collection and returns the respective Connection.
114  Connection add(const ValueType& listener)
115  {
116  // Actual insertion
117  mListeners.push_back(listener);
118 
119  // Let the Listener know about its container and iterator
120  Iterator added = mListeners.end() - 1;
121  added->setEnvironment(*this, added);
122 
123  // Create connection from the added Listener
124  return added->shareConnection();
125  }
126 
127  // Removes a listener through the given iterator.
128  void remove(Iterator iterator)
129  {
130  // Erase in O(1)
131  iterator->swap(mListeners.back());
132  mListeners.pop_back();
133  }
134 
135  // Removes all listeners from the container
136  void clear()
137  {
138  mListeners.clear();
139  }
140 
141  // Invokes all stored functions with arg as argument.
142  void call(Parameter arg) const
143  {
144  AURORA_FOREACH(const ValueType& listener, mListeners)
145  listener.call(arg);
146  }
147 
148  private:
149  Container mListeners;
150  };

153  // Associative container (map) for listener
154  template <typename Trigger, typename Parameter>
155  class ListenerMap
156  {
157  public:
158  // The type of the function together with the id
159  typedef Listener<Parameter> ValueType;
160 
161  // The event identifier associated with the listener
162  typedef Trigger KeyType;
163 
164  private:
165  // The container type used to store the callback functions
166  typedef std::multimap<KeyType, ValueType> Container;
167 
168  // The const iterator type (used internally)
169  typedef typename Container::const_iterator ConstIterator;
170 
171  public:
172  // The iterator type (used to disconnect listeners)
173  typedef typename Container::iterator Iterator;
174 
175  public:
176  // Inserts a new listener to the collection and returns the respective Connection.
177  Connection add(const KeyType& trigger, const ValueType& listener)
178  {
179  // Actual insertion
180  Iterator added = mListeners.insert( std::make_pair(trigger, listener) );
181 
182  // Let the Listener know about its container and iterator
183  added->second.setEnvironment(*this, added);
184 
185  // Create connection from the added Listener
186  return added->second.shareConnection();
187  }
188 
189  // Removes a listener through the given iterator.
190  void remove(Iterator iterator)
191  {
192  mListeners.erase(iterator);
193  }
194 
195  // Removes all listeners for a specific key
196  void clear(KeyType key)
197  {
198  mListeners.erase(key);
199  }
200 
201  // Removes all listeners from the container
202  void clearAll()
203  {
204  mListeners.clear();
205  }
206 
207  // Invokes all stored functions with arg as argument.
208  void call(Trigger event, Parameter arg) const
209  {
210  std::pair<ConstIterator, ConstIterator> range = mListeners.equal_range(event);
211 
212  for (ConstIterator itr = range.first; itr != range.second; ++itr)
213  itr->second.call(arg);
